>>> MyList = __import__('1-my_list').MyList

# création + append + impression normale
>>> ml = MyList()
>>> ml.append(1); ml.append(4); ml.append(2); ml.append(3); ml.append(5)
>>> print(ml)
[1, 4, 2, 3, 5]

# print_sorted affiche trié sans modifier l'original
>>> ml.print_sorted()
[1, 2, 3, 4, 5]
>>> print(ml)
[1, 4, 2, 3, 5]

# doublons
>>> ml2 = MyList([3, 1, 3, 2])
>>> ml2.print_sorted()
[1, 2, 3, 3]

# négatifs
>>> ml3 = MyList([0, -2, 5, -1])
>>> ml3.print_sorted()
[-2, -1, 0, 5]

# vide
>>> ml4 = MyList()
>>> ml4.print_sorted()
[]

# MyList hérite bien de list
>>> isinstance(ml, list)
True
